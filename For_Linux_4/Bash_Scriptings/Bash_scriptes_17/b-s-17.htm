Уыртр 16. Ях№хэря№ртыхэшх ттюфр/тћтюфр

Т ёшёђхьх яю-ѓьюыїрэшў тёхуфр юђъ№ћђћ ђ№ш "єрщыр" -- stdin (ъыртшрђѓ№р), stdout (§ъ№рэ) ш stderr (тћтюф ёююсљхэшщ юс юјшсърѕ эр §ъ№рэ). нђш, ш ыўсћх ф№ѓушх юђъ№ћђћх єрщыћ, ьюуѓђ сћђќ ях№хэря№тыхэћ. Т фрээюь ёыѓїрх, ђх№ьшэ "ях№хэря№ртыхэшх" ючэрїрхђ яюыѓїшђќ тћтюф шч єрщыр, ъюьрэфћ, я№юу№рььћ, ёіхэр№шџ шыш фрцх юђфхыќэюую сыюър т ёіхэр№шш (ёь. Я№шьх№ 3-1 ш Я№шьх№ 3-2) ш ях№хфрђќ хую эр тѕюф т ф№ѓующ єрщы, ъюьрэфѓ, я№юу№рььѓ шыш ёіхэр№шщ.

б ърцфћь юђъ№ћђћь єрщыюь ётџчрэ фхёъ№шяђю№ єрщыр. [1] Фхёъ№шяђю№ћ єрщыют stdin, stdout ш stderr -- 0, 1 ш 2, ёююђтхђёђтхээю. Я№ш юђъ№ћђшш фюяюыэшђхыќэћѕ єрщыют, фхёъ№шяђю№ћ ё 3 яю 9 юёђрўђёџ эхчрэџђћьш. Шэюуфр фюяюыэшђхыќэћх фхёъ№шяђю№ћ ьюуѓђ ёюёыѓцшђќ эхяыюѕѓў ёыѓцсѓ, т№хьхээю ёюѕ№рэџџ т ёхсх ёёћыъѓ эр stdin, stdout шыш stderr. [2] нђю ѓя№юљрхђ тючт№рђ фхёъ№шяђю№ют т эю№ьрыќэюх ёюёђюџэшх яюёых ёыюцэћѕ ьрэшяѓыџішщ ё ях№хэря№ртыхэшхь ш ях№хёђрэютърьш (ёь. Я№шьх№ 16-1).

   COMMAND_OUTPUT >

      # Ях№хэря№ртыхэшх stdout (тћтюфр) т єрщы.

      # Хёыш єрщы юђёѓђёђтютры, ђю юэ ёючфхђёџ, шэрїх -- ях№хчряшёћтрхђёџ.

 

      ls -lR > dir-tree.list

      # бючфрхђ єрщы, ёюфх№црљшщ ёяшёюъ фх№хтр ърђрыюуют.

 

   : > filename

      # Юях№рішџ > ѓёхърхђ єрщы "filename" фю эѓыхтющ фышэћ.

      # Хёыш фю тћяюыэхэшџ юях№рішш єрщыр эх ёѓљхёђтютрыю,

      # ђю ёючфрхђёџ эютћщ єрщы ё эѓыхтющ фышэющ (ђюђ цх §єєхъђ фрхђ ъюьрэфр 'touch').

      # бшьтюы : тћёђѓярхђ чфхёќ т №юыш ьхёђючряюыэшђхыџ, эх тћтюфџ эшїхую.

 

   > filename

      # Юях№рішџ > ѓёхърхђ єрщы "filename" фю эѓыхтющ фышэћ.

      # Хёыш фю тћяюыэхэшџ юях№рішш єрщыр эх ёѓљхёђтютрыю,

      # ђю ёючфрхђёџ эютћщ єрщы ё эѓыхтющ фышэющ (ђюђ цх §єєхъђ фрхђ ъюьрэфр 'touch').

      # (ђюђ цх №хчѓыќђрђ, їђю ш тћјх -- ": >", эю §ђюђ тр№шрэђ эх№рсюђюёяюёюсхэ

      # т эхъюђю№ћѕ ъюьрэфэћѕ юсюыюїърѕ.)

 

   COMMAND_OUTPUT >>

      # Ях№хэря№ртыхэшх stdout (тћтюфр) т єрщы.

      # бючфрхђ эютћщ єрщы, хёыш юэ юђёѓђёђтютры, шэрїх -- фюяшёћтрхђ т ъюэхі єрщыр.

 

 

      # Юфэюёђ№юїэћх ъюьрэфћ ях№хэря№ртыхэшџ

      # (чрђ№руштрўђ ђюыќъю ђѓ ёђ№юъѓ, т ъюђю№ющ юэш тёђ№хїрўђёџ):

      # --------------------------------------------------------------------

 

   1>filename

      # Ях№хэря№ртыхэшх тћтюфр (stdout) т єрщы "filename".

   1>>filename

      # Ях№хэря№ртыхэшх тћтюфр (stdout) т єрщы "filename", єрщы юђъ№ћтрхђёџ т №хцшьх фюсртыхэшџ.

   2>filename

      # Ях№хэря№ртыхэшх stderr т єрщы "filename".

   2>>filename

      # Ях№хэря№ртыхэшх stderr т єрщы "filename", єрщы юђъ№ћтрхђёџ т №хцшьх фюсртыхэшџ.

   &>filename

      # Ях№хэря№ртыхэшх stdout ш stderr т єрщы "filename".

 

      #==============================================================================

      # Ях№хэря№ртыхэшх stdout, ђюыќъю фыџ юфэющ ёђ№юъш.

      LOGFILE=script.log

 

      echo "нђр ёђ№юър сѓфхђ чряшёрэр т єрщы \"$LOGFILE\"." 1>$LOGFILE

      echo "нђр ёђ№юър сѓфхђ фюсртыхэр т ъюэхі єрщыр \"$LOGFILE\"." 1>>$LOGFILE

      echo "нђр ёђ№юър ђюцх сѓфхђ фюсртыхэр т ъюэхі єрщыр \"$LOGFILE\"." 1>>$LOGFILE

      echo "нђр ёђ№юър сѓфхђ тћтхфхэр эр §ъ№рэ ш эх яюярфхђ т єрщы \"$LOGFILE\"."

      # Яюёых ърцфющ ёђ№юъш, ёфхырээюх ях№хэря№ртыхэшх ртђюьрђшїхёъш "ёс№рёћтрхђёџ".

 

 

 

      # Ях№хэря№ртыхэшх stderr, ђюыќъю фыџ юфэющ ёђ№юъш.

      ERRORFILE=script.errors

 

      bad_command1 2>$ERRORFILE       #  бююсљхэшх юс юјшсъх чряшјхђёџ т $ERRORFILE.

      bad_command2 2>>$ERRORFILE      #  бююсљхэшх юс юјшсъх фюсртшђёџ т ъюэхі $ERRORFILE.

      bad_command3                    #  бююсљхэшх юс юјшсъх сѓфхђ тћтхфхэю эр stderr,

                                      #+ ш эх яюярфхђ т $ERRORFILE.

      # Яюёых ърцфющ ёђ№юъш, ёфхырээюх ях№хэря№ртыхэшх ђръцх ртђюьрђшїхёъш "ёс№рёћтрхђёџ".

      #==============================================================================

 

 

 

   2>&1

      # Ях№хэря№ртыџхђёџ stderr эр stdout.

      # бююсљхэшџ юс юјшсърѕ ях№хфрўђёџ ђѓфр цх, ъѓфр ш ёђрэфр№ђэћщ тћтюф.

 

   i>&j

      # Ях№хэря№ртыџхђёџ єрщы ё фхёъ№шяђю№юь i т j.

      # Тћтюф т єрщы ё фхёъ№шяђю№юь i ях№хфрхђёџ т єрщы ё фхёъ№шяђю№юь j.

 

   >&j

      # Ях№хэря№ртыџхђёџ  єрщы ё фхёъ№шяђю№юь 1 (stdout) т єрщы ё фхёъ№шяђю№юь j.

      # Тћтюф эр stdout ях№хфрхђёџ т єрщы ё фхёъ№шяђю№юь j.

 

   0< FILENAME

    < FILENAME

      # Ттюф шч єрщыр.

      # Яр№эрџ ъюьрэфх ">", їрёђю тёђ№хїрхђёџ т ъюьсшэрішш ё эхщ.

      #

      # grep search-word <filename

 

 

   [j]<>filename

      # дрщы "filename" юђъ№ћтрхђёџ эр їђхэшх ш чряшёќ, ш ётџчћтрхђёџ ё фхёъ№шяђю№юь "j".

      # Хёыш "filename" юђёѓђёђтѓхђ, ђю юэ ёючфрхђёџ.

      # Хёыш фхёъ№шяђю№ "j" эх ѓърчрэ, ђю, яю-ѓьюыїрэшў, сх№хфёџ фхёъ№шяђю№ 0, stdin.

      #

      # Ъръ юфэю шч я№шьхэхэшщ §ђюую -- чряшёќ т ъюэъ№хђэѓў яючшішў т єрщых.

      echo 1234567890 > File    # Чряшёрђќ ёђ№юъѓ т єрщы "File".

      exec 3<> File       # Юђъ№ћђќ "File" ш ётџчрђќ ё фхёъ№шяђю№юь 3.

      read -n 4 <&3             # Я№юїшђрђќ 4 ёшьтюыр.

      echo -n . >&3             # Чряшёрђќ ёшьтюы ђюїъш.

      exec 3>&-                 # Чръ№ћђќ фхёъ№шяђю№ 3.

      cat File                  # ==> 1234.67890

      # Я№юшчтюыќэћщ фюёђѓя, фр ш ђюыќъю!

 

 

 

   |

      # Ъюэтхщх№ (ърэры).

      # гэштх№ёрыќэюх ё№хфёђтю фыџ юсњхфшэхэшџ ъюьрэф т юфэѓ іхяюїъѓ.

      # Яюѕюцх эр ">", эю эр ёрьюь фхых -- сюыхх юсјш№эрџ.

      # Шёяюыќчѓхђёџ фыџ юсњхфшэхэшџ ъюьрэф, ёіхэр№шхт, єрщыют ш я№юу№рьь т юфэѓ іхяюїъѓ (ъюэтхщх№).

      cat *.txt | sort | uniq > result-file

      # бюфх№цшьюх тёхѕ єрщыют .txt ёю№ђш№ѓхђёџ, ѓфрыџўђёџ яютђю№џўљшхёџ ёђ№юъш,

      # №хчѓыќђрђ ёюѕ№рэџхђёџ т єрщых "result-file".

Юях№рішш ях№хэря№ртыхэшџ ш/шыш ъюэтхщх№ћ ьюуѓђ ъюьсшэш№ютрђќёџ т юфэющ ъюьрэфэющ ёђ№юъх.

Фюяюыэшђхыќэћх ётхфхэшџ ю ях№хэря№ртыхэшш ттюфр/тћтюфр тћ эрщфхђх т Я№шыюцхэшх D.

16.1. б яюьюљќў ъюьрэфћ exec

Ъюьрэфр exec <filename ях№хэря№ртыџхђ ттюф ёю stdin эр єрщы. б §ђюую ьюьхэђр тхёќ ттюф, тьхёђю stdin (юсћїэю §ђю ъыртшрђѓ№р), сѓфхђ я№юшчтюфшђќёџ шч §ђюую єрщыр. нђю фрхђ тючьюцэюёђќ їшђрђќ ёюфх№цшьюх єрщыр, ёђ№юъѓ чр ёђ№юъющ, ш рэрышчш№ютрђќ ърцфѓў ттхфхээѓў ёђ№юъѓ ё яюьюљќў sed ш/шыш awk.

Я№шьх№ 16-1. Ях№хэря№ртыхэшх stdin ё яюьюљќў exec

#!/bin/bash

# Ях№хэря№ртыхэшх stdin ё яюьюљќў 'exec'.

 

 

exec 6<&0          # бтџчрђќ фхёъ№. #6 ёю ёђрэфр№ђэћь ттюфюь (stdin).

                   # бюѕ№рэџџ stdin.

 

exec < data-file   # stdin чрьхэџхђёџ єрщыюь "data-file"

 

read a1            # зшђрхђёџ ях№трџ ёђ№юър шч "data-file".

read a2            # зшђрхђёџ тђю№рџ ёђ№юър шч "data-file."

 

echo

echo "быхфѓўљшх ёђ№юъш сћыш я№юїшђрэћ шч єрщыр."

echo "-----------------------------------------"

echo $a1

echo $a2

 

echo; echo; echo

 

exec 0<&6 6<&-

#  Тюёёђрэртыштрхђёџ stdin шч фхёъ№. #6, уфх юэ сћы я№хфтр№шђхыќэю ёюѕ№рэхэ,

#+ ш фхёъ№. #6 чръ№ћтрхђёџ ( 6<&- ) юётюсюцфрџ хую фыџ ф№ѓушѕ я№юіхёёют.

#

# <&6 6<&-    фрхђ ђюђ цх №хчѓыќђрђ.

 

echo -n "Ттхфшђх ёђ№юъѓ  "

read b1  # вхях№ќ єѓэъішџ "read", ъръ ш ёыхфютрыю юцшфрђќ, я№шэшьрхђ фрээћх ё юсћїэюую stdin.

echo "бђ№юър, я№шэџђрџ ёю stdin."

echo "--------------------------"

echo "b1 = $b1"

 

echo

 

exit 0

Рэрыюушїэю, ъюэёђ№ѓъішџ exec >filename ях№хэря№ртыџхђ тћтюф эр stdout т чрфрээћщ єрщы. Яюёых §ђюую, тхёќ тћтюф юђ ъюьрэф, ъюђю№ћщ юсћїэю эря№ртыџхђёџ эр stdout, ђхях№ќ тћтюфшђёџ т §ђюђ єрщы.

Я№шьх№ 16-2. Ях№хэря№ртыхэшх stdout ё яюьюљќў exec

#!/bin/bash

# reassign-stdout.sh

# ----------------------------------------------------------- #

# Тхёќ тћтюф юђ ъюьрэф, т фрээюь сыюъх, чряшёћтрхђёџ т єрщы $LOGFILE.

echo "-------------------------------------"

echo

echo

echo; echo

echo "Тћтюф ъюьрэфћ \"df\""

echo

df

echo

echo

Я№шьх№ 16-3. Юфэют№хьхээюх ях№хэря№ртыхэшх ѓёђ№ющёђт, stdin ш stdout, ё яюьюљќў ъюьрэфћ exec

#!/bin/bash

# upperconv.sh

# Я№хюс№рчютрэшх ёшьтюыют тю тѕюфэюь єрщых т тх№ѕэшщ №хушёђ№.

 

E_FILE_ACCESS=70

E_WRONG_ARGS=71

 

if [ ! -r "$1" ]     # дрщы фюёђѓяхэ фыџ їђхэшџ?

then

  echo "Эхтючьюцэю я№юїшђрђќ шч чрфрээюую єрщыр!"

  echo "Яю№џфюъ шёяюыќчютрэшџ: $0 input-file output-file"

  exit $E_FILE_ACCESS

fi                   #  Т ёыѓїрх, хёыш тѕюфэющ єрщы ($1) эх чрфрэ

                     #+ ъюф чртх№јхэшџ сѓфхђ §ђшь цх.

 

if [ -z "$2" ]

then

  echo "Эхюсѕюфшью чрфрђќ тћѕюфэющ єрщы."

  echo "Яю№џфюъ шёяюыќчютрэшџ: $0 input-file output-file"

  exit $E_WRONG_ARGS

fi

 

 

exec 4<&0

exec < $1            # Эрчэрїшђќ ттюф шч тѕюфэюую єрщыр.

exec > $2            # Эрчэрїшђќ тћтюф т тћѕюфэющ єрщы.

                     # Я№хфяюырурхђёџ, їђю тћѕюфэющ єрщы фюёђѓяхэ фыџ чряшёш

                     # (фюсртшђќ я№ютх№ъѓ?).

 

# -----------------------------------------------
