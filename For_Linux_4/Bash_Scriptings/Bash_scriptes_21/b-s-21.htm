--------------

#!/bin/sh

    ;;

    echo "Юђърч юђ ттюфр.";;

  255)

    if test -s $tempfile ; then

      cat $tempfile

    else

      echo "Эрцрђр ъыртшјр ESC."

    fi

    ;;

   нђр я№юу№рььр эхьэюую ёыюцэхх, їхь я№хфћфѓљрџ. быхфѓўљшх ёђ№юъш

   юя№хфхыџўђ т№хьхээћщ єрщы ш хую ѓфрыхэшх я№ш чртх№јхэшш я№юу№рььћ:

        trap "rm -f $tempfile" 0 1 2 5 15

   яю§ђюьѓ я№шјыюёќ хИ чрьхэшђќ эр mktemp. еюђџ ёъ№шяђ №рсюђрђќ сѓфхђ т

   ыўсюь ёыѓїрх, ђ.ъ. т ёыѓїрх юјшсъш ъюэёђ№ѓъішџ || яючтюыџхђ

   ёєю№ьш№ютрђќ шьџ єрщыр тшфр /tmp/test$$, уфх $$ -- чэрїхэшх ухэх№рђю№р

   ёыѓїрщэћѕ їшёхы. еюђџ, эр ьющ тчуыџф, ъю№№хъђэхщ сћыю сћ шёяюыќчютрђќ

   ях№хьхээѓў $RANDOM. -- я№шь. Р.Ъ.)

 

   Т ях№тющ ёђ№юъх фхырхђёџ яюяћђър ёючфрђќ т№хьхээћщ єрщы ё яюьюљќў

   ѓђшышђћ mktemp. Хёыш §ђю эх яюыѓїрхђёџ, юэ ёючфрхђёџ т№ѓїэѓў, т
