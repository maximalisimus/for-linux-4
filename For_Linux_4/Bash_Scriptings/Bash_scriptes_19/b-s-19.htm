#  Я№хюс№рчютрэшх тёхѕ єрщыют т чрфрээюь  ърђрыюух,
#+ шч у№рєшїхёъюую єю№ьрђр MacPaint, т єю№ьрђ "pbm".
 
#  Шёяюыќчѓхђёџ ѓђшышђр "macptopbm", тѕюфџљрџ т ёюёђрт яръхђр "netpbm",
#+ ъюђю№ћщ ёюя№ютюцфрхђёџ Brian Henderson (bryanh@giraffe-data.com).
#  Netpbm -- ёђрэфр№ђэћщ яръхђ фыџ сюыќјшэёђтр фшёђ№шсѓђштют Linux.
 
OPERATION=macptopbm
SUFFIX=pbm          # Эютюх №рёјш№хэшх єрщыр.
 
if [ -n "$1" ]
then
  directory=$1      # Хёыш ърђрыюу чрфрэ т ъюьрэфэющ ёђ№юъх я№ш тћчютх ёіхэр№шџ
else
  directory=$PWD    # Шэрїх я№юёьрђ№штрхђёџ ђхъѓљшщ ърђрыюу.
fi
 
#  Тёх єрщыћ т ърђрыюух, шьхўљшх №рёјш№хэшх ".mac", ёїшђрўђёџ єрщырьш
#+ єю№ьрђр  MacPaint.
 
for file in $directory/* # Яюфёђрэютър шьхэ єрщыют.
do
  filename=${file%.*c}   #  гфрышђќ №рёјш№хэшх ".mac" шч шьхэш єрщыр
                         #+ ( ё јрсыюэюь '.*c' ёютярфрўђ тёх яюфёђ№юъш
                         #+ эрїшэрўљшхёџ ё '.' ш чрърэїштрўљшхёџ 'c',
  $OPERATION $file > "$filename.$SUFFIX"
                         # Я№хюс№рчютрэшх ё ях№хэря№ртыхэшхь т єрщы ё эютћь шьхэхь
  rm -f $file            # гфрыхэшх ю№шушэрыќэюую єрщыр яюёых я№хюс№рчютрэшџ.
  echo "$filename.$SUFFIX"  # Тћтюф эр stdout.
done
 
exit 0
